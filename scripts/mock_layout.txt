# Mock provider script focused on display/layout features. Where
# mock_demo.txt walks through the rendering pipeline end-to-end (head
# preview, head+tail elision, diff coloring, etc.), this one is the
# place to add fixtures for layout-specific behaviors that are easy to
# eyeball but tedious to cover in unit tests — header reflow, tool-
# output gutter rendering (incl. tab expansion), and streaming-markdown
# wrap.
#
#   HAX_PROVIDER=mock HAX_MOCK_SCRIPT=scripts/mock_layout.txt build/hax
#
# Resize the terminal to a known width before running for predictable
# break points (the comments below assume a 100-cell window — the
# default display_width() cap). DSL reference lives in mock_demo.txt.

# cmd_classify routes pure read-only invocations (ls, find, grep, …)
# through the silent header — single dim line, no reflow target. To
# exercise the verbose multi-row reflow we need commands that classify
# as non-exploration (shell control flow, mutating ops, or unknown
# binaries are reliable triggers).

# ---------- turn 1: bash header, fits on one row ----------------------
# Short verbose command — no wrap, no ellipsis. The "for d in …; do …;
# done" wrapper trips cmd_classify into the verbose path.
text Short bash command first, for a baseline.
tool bash {"command":"for d in src; do echo $d; done"}
usage in=80 out=10
end-turn

# ---------- turn 2: bash header, reflows across two rows --------------
# Long enough that the first row's "[bash] " prefix forces a word-
# boundary break, but short enough to fit in the 3-row budget so no
# truncation kicks in.
text Medium-length bash command — should wrap across two rows.
tool bash {"command":"for f in src/providers/openai.c src/providers/codex.c src/providers/llamacpp.c; do echo $f; done"}
usage in=160 out=18
end-turn

# ---------- turn 3: bash header, fills 3 rows exactly -----------------
# Long enough to fill three rows but stops just under the budget — no
# trailing "...". Demonstrates the upper edge of the reflow window.
text Long bash command — should fill all 3 rows without truncation.
tool bash {"command":"for f in src/providers/openai.c src/providers/codex.c src/providers/llamacpp.c src/providers/openrouter.c src/providers/openai_compat.c src/providers/mock.c; do echo \"== $f ==\"; wc -l \"$f\"; head -1 \"$f\"; tail -1 \"$f\"; done"}
usage in=260 out=20
end-turn

# ---------- turn 4: bash header, overflows 3 rows + truncates ---------
# Pushes well past the 3-row budget so the last row visibly ends in
# "..." with a clean word-boundary cut.
text Very long bash command — should overflow and truncate.
tool bash {"command":"for f in src/providers/openai.c src/providers/codex.c src/providers/llamacpp.c src/providers/openrouter.c src/providers/openai_compat.c src/providers/mock.c src/providers/probe.c; do echo \"== $f ==\"; wc -l \"$f\"; head -3 \"$f\"; tail -3 \"$f\"; echo \"-- lines summary --\"; awk '{c++} END {print c}' \"$f\"; echo \"-- words summary --\"; awk '{w+=NF} END {print w}' \"$f\"; done"}
usage in=320 out=24
end-turn

# ---------- turn 5: bash output with long lines (per-row truncation) ---
# Generates a mix of short and very-wide lines so the per-row cell cap
# in tool_render kicks in: each long line should land as truncated
# content with a trailing "..." inside the dim block. The for-loop
# wrapper keeps this on the verbose path; the strip glyphs (┌ / │ / └)
# anchor the gutter on every preview row.
text Long-line bash output — should show per-row "..." truncation.
tool bash {"command":"for i in 1 2 3; do printf 'X%.0s' $(seq 1 200); echo; printf 'short line\\n'; done"}
usage in=380 out=14
end-turn

# ---------- turn 6: leading tabs in tool output (gutter integrity) ----
# Lines with leading tabs used to wrap out of the gutter strip:
# wcwidth('\t') == -1 left cell math believing a tab took 1 cell while
# the terminal expanded it to the next column-multiple-of-8 tab stop,
# pushing content past content_budget. tool_render now substitutes \t
# with 4 spaces before content reaches truncate_for_display, so this
# block should render with a clean ┌ / │ / └ gutter, tabs visualized as
# 4-space indents, and per-row "..." on the long line.
text Tab-laden output — gutter should stay intact and tabs render as 4 spaces.
tool bash {"command":"printf '\\t\\t// a long comment with enough text after the leading tabs that the row needs per-cell truncation at the right margin\\n\\t\\tlong_function_call(first_argument, second_argument, third_argument);\\n\\thello from a less-indented line\\nno-tab line at column zero\\n'"}
usage in=400 out=18
end-turn

# ---------- turn 7: read silent header --------------------------------
# Read goes through the silent path: the bracketed `[read]` plus the
# basename + ":N-M" extra render as a single dim line. Included here
# so the layout demo also covers the silent header for visual
# comparison against the verbose bash blocks above; the verbose
# multi-row reflow won't apply until a non-silent path tool lands.
text Read with a deeply nested path.
tool read {"path":"src/providers/this/is/an/intentionally/very/deeply/nested/imaginary/path/to/some/test_fixture.c","offset":30,"limit":31}
usage in=400 out=12
end-turn

text Done — that covers verbose-header reflow, per-row truncation, tab expansion, and the silent path.
usage in=440 out=14
end-turn

# ---------- turn 15: code fence with tab-indented source --------------
# Models emitting source inside a fenced block sometimes use real tabs.
# Without substitution those reach the terminal raw and jump to its
# native tab stops, producing inconsistent indent depth depending on
# what precedes the tab on the line. emit_text now substitutes \t → 4
# spaces before either branch (verbatim-bypass or wrap engine), so
# every indent level below should look like exactly 4 spaces.
text Tab-indented code fence — indent depth should be consistent (4 spaces per level):
text \n```c
text \nint main(int argc, char **argv)

text \n\tif (argc > 1) {
text \n\t\tprintf("hello %s\\n", argv[1]);
text \n\t}
text \n\treturn 0;
text \n}
text \n```

end-turn

# ---------- turn 19: GFM table, model emits one row per delta --------
# Each row arrives in a separate delta, so this exercises the cross-feed
# table buffering: the renderer can't lay out columns until it has seen
# every row, so it diverts lines into table_buf from the `|`-led header
# (confirmed a table by the `|---|` delimiter on the next line) until the
# blank line / non-pipe line that ends the block, then renders an aligned
# grid — bold header, dim `│` column separators, a `─┼─` underline, and
# padded columns.
text Here's a small table:
text \n| Tool | What it does |
text \n|---|---|
text \n| read | Read files from disk |
text \n| bash | Run shell commands via /bin/sh -c |
usage in=940 out=30
end-turn
